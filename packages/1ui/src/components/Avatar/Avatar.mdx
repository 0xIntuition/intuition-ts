import { Canvas, Meta, ArgTypes, Title } from '@storybook/blocks'

import * as AvatarStories from './Avatar.stories.tsx'

<Title />

An image element with a fallback for representing the user.

<Meta of={AvatarStories} />

### Default

<Canvas of={AvatarStories.Default} sourceState="none" />

```tsx
<Avatar>
  <AvatarImage src="https://github.com/shadcn.png" alt="@shadcn" />
  <AvatarFallback>CN</AvatarFallback>
</Avatar>
```

### Fallback

If `AvatarImage` is unable to retrieve an image, `AvatarFallback` will be shown.

<Canvas of={AvatarStories.Fallback} sourceState="none" />

```tsx
<Avatar>
  <AvatarImage src="broken-link" />
  <AvatarFallback>CN</AvatarFallback>
</Avatar>
```
