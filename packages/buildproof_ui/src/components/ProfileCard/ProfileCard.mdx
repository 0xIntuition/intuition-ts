#### The ProfileCard component displays an identity profile information with various details like role, followers, bio, and more.

<Source
  dark
  language="tsx"
  code={`
import { ProfileCard } from '@0xintuition/buildproof_ui'

<ProfileCard
  variant="user"
  avatarSrc="path/to/avatar.jpg"
  name="John Doe"
  id="0x1234567890abcdef1234567890abcdef12345678"
  roles={['Developer', 'Judge']}
  stats={{
    numberOfFollowers: 123,
    numberOfFollowing: 45,
  }}
  ipfsLink="https://example.com"
  bio="John Doe is a blockchain enthusiast. He loves to learn new things and share his knowledge with others. He is also a contributor to various open-source projects."
>
  <Button
    variant="accent"
    onClick={() => window.open('https://example.com', '_blank')}
  >
    Follow
  </Button>
</ProfileCard>
`} />

<Canvas of={Stories.BasicUsage} sourceState="none" />
